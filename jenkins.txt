                } 
            } 
        } 
        stage('Verify Files') { 
            steps { 
                bat 'dir /S /B' 
            } 
        } 
        stage('Deploy') { 
            steps {
                script { 
                    def srcPath = "C:\\Users\\Shravani Kijbile\\eclipse-workspace\\Devops\\JenkinsDemo\\src\\main\\webapp" 
                    def destPath = "C:\\Program Files\\Apache Software Foundation\\apache-tomcat-11.0.14\\webapps\\Devops"

        } 
    }
